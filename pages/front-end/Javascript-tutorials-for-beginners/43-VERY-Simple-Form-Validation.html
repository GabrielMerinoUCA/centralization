XLIII. VERY Simple Form Validation

Javascript allows to make form validation from the client side. Client side form validation can be very useful to provide feedback to the user about simple mistakes or restrictions without having to overload the server with unnecessary requests. However, if there is any sort of complex form validation, do not use Javascript.

index.html



<body>

     <form name="form1">

          <input type="text" name="name" /><br>

          <div id="message" style="color: red;"></div>

          <input type="checkbox" name="biking" value="biking"/> Biking<br>

          <input type="checkbox" name="skiing" value="skiing"/> Skiing<br>

          <input type="checkbox" name="driving" value="driving"/> Driving<br>

          <select name="color">

               <option>Red</option>

               <option>Green</option>

               <option>Blue</option>

          </select>

          <input type="submit" name="submit value="Submit />

    </form>

</body>


test.js



let myForm = document.forms.form1;

let message = document.getElementById( "message" );


/* Execute this function when the form is submitted */

myForm.onsubmit = function() {
     if(myForm.name.value === "") {
          message.contentText = "Field cannot be empty!";
          /* returning false on the onsubmit event function will prevent the submission of the form */
          return false;
     } else {
          message.contentText = "";
          /* And returning true will allow form submission */
          return true;
     }
}

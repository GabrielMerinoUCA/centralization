XII. Else if Statements

Ok so nothing fancy, the syntax is very conventional but I'll still show it

test.js



var age = 12;

if(age > 30) {
    console.log( "Age is over 30" );
}

else if(age > 20) {
    console.log( "Age is over 20" );
}

// will be executed

else {
    console.log( "Age is under 20" );
}

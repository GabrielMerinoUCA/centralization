VIII. Class Inheritance

Javascript supports Inheritance like many other programming languages and is very similar to them syntax wise.

main.js



class User {
     constructor(email, name) {
          this.email = email;
          this.name = name;
     }

     login() {
          console.log(this.email + " logged in");
          return this;
     }

     logout() {
          console.log(this.email + " logged out");
          return this;
     }
}

class Admin extends User {
     deleteUser(user, users) {
          for(let i = 0; i < users.length; i++) {
               if(user = user[i]) {
                    // delete 1 element after index 'i' (could be more)
                         users.splice(i, 1)
               }
          }
     }
}


let users = [newUser("john@gmail.com", "John Doe"), new User("anthony@gmail.com", "Anthony Dubois")];

let admin1 = new Admin("maximus@gmail.com", "Maximus Optimus");


// Javascript sends all arguments by reference so you can change the origin this way

admin1.deleteUser(users[0], users);

// John Doe got deleted from the array
